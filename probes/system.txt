[ abort ]

Construct executable that performs abort in a function.

/__init__.py/#!/pl/python
	__factor_domain__ = 'system'
	__factor_type__ = 'executable'

/src/
	/program.c/#!/pl/c
		#include <stdio.h>
		void
		sub(void)
		{
			abort();
		}

		int
		main(int argc, char *argv[])
		{
			sub();
			return(1);
		}
